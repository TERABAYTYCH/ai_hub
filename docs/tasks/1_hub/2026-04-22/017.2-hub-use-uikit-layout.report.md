## Что сделано

### 1. Создан Layout.tsx в ui-kit

Создан универсальный компонент `Layout` в `libs/ui-kit/src/components/layout/Layout.tsx`:

```typescript
export function Layout({
  children,
  serviceName = 'Ject Hub',
  staticMenuItems = [],
}: {
  children: ReactNode;
  serviceName?: string;
  staticMenuItems?: MenuItem[];
}) {
  const { user, logout } = useAuth();
  const { manifests } = useMicroserviceManifests();
  const username = user?.username || user?.email || 'User';

  // Динамические меню из manifest.json
  const serviceMenuItems: MenuItem[] = manifests.map((m) => ({
    id: m.serviceId,
    label: m.name,
    icon: 'bi bi-grid',
    children: m.navigation.map((n) => ({
      id: `${m.serviceId}-${n.path}`,
      label: n.label,
      icon: n.icon,
      path: n.path,
    })),
  }));

  const menuItems: MenuItem[] = [...staticMenuItems, ...serviceMenuItems];

  return (
    <AppLayout
      menuItems={menuItems}
      serviceName={serviceName}
      username={username}
      onLogout={logout}
    >
      {children}
    </AppLayout>
  );
}
```

**Логика:**
- Принимает `staticMenuItems` — статические пункты меню текущего сервиса
- Автоматически загружает manifest.json из всех зарегистрированных сервисов
- Комбинирует статические и динамические меню
- Использует `useAuth()` для username и logout
